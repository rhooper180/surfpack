LoadData[name = roseData, file = 'rose100.txt']
CreateSurface[name = krigsurf, type = 'Kriging', data = roseData, theta_vars = (10,10)]
CreateSurface[name = marssurf, type = 'Mars', data = roseData]
CreateSurface[name = rbfsurf, type = 'RBFNet', data = roseData]
AxesBounds[name = densedef, file = 'ndef']
GridPoints[def = densedef, name = denseRoseData, test_function = 'rosenbrock']
Evaluate[surface = krigsurf, input_data = denseRoseData]
Evaluate[surface = marssurf, input_data = denseRoseData]
Evaluate[surface = rbfsurf, input_data = denseRoseData]
SaveData[data = denseRoseData, file = 'results.txt']
